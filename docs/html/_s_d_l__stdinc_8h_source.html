497/* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */
498SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)
499{
500#ifdef __APPLE__
501 memset_pattern4(dst, &val, dwords * 4);
502#elif defined(__GNUC__) && defined(__i386__)
503 int u0, u1, u2;
504 __asm__ __volatile__ (
505 "cld \n\t"
506 "rep ; stosl \n\t"
507 : "=&D" (u0), "=&a" (u1), "=&c" (u2)
508 : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))
509 : "memory"
510 );
511#else
512 size_t _n = (dwords + 3) / 4;
513 Uint32 *_p = SDL_static_cast(Uint32 *, dst);
514 Uint32 _val = (val);
515 if (dwords == 0) {
516 return;
517 }
518 switch (dwords % 4) {
519 case 0: do { *_p++ = _val; SDL_FALLTHROUGH;
520 case 3: *_p++ = _val; SDL_FALLTHROUGH;
521 case 2: *_p++ = _val; SDL_FALLTHROUGH;
522 case 1: *_p++ = _val;
523 } while ( --_n );
524 }
525#endif
526}
